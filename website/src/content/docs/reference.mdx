  sonarrFields,
  jellyfinFields,
  jellyseerrFields,
  stateFields,
  operatorDefinitions,
  durationSyntax,
} from "../../../../src/config/field-registry";

export const code = (s) => `<code>${s}</code>`;

export const durationRows = () =>
  durationSyntax.map((unit) => [
    unit.label,
    code(unit.suffix),
    code(unit.example),
  ]);

export const typeCompatRows = () => {
  const typeToOperators = new Map();
  for (const [name, def] of Object.entries(operatorDefinitions)) {
    for (const type of def.compatibleTypes) {
      const existing = typeToOperators.get(type) ?? [];
      existing.push(name);
      typeToOperators.set(type, existing);
    }
  }
  const typeOrder = ["string", "number", "boolean", "date", "array"];
  return typeOrder.map((type) => {
    const ops = typeToOperators.get(type) ?? [];
    return [code(type), ops.map((o) => code(o)).join(", ")];
  });
};

### Null handling

- `older_than` **always matches** null dates — guard with an additional condition if needed:

  ```yaml
  # Without guard: matches movies with no digital release date (null = always matches)
  - field: radarr.digital_release
    operator: older_than
    value: 6mo

  # With guard: only matches movies that have a digital release date AND it's old
  conditions:
    operator: AND
    children:
      - field: radarr.has_file
        operator: equals
        value: true
      - field: radarr.digital_release
        operator: older_than
        value: 6mo
  ```

- `newer_than` **never matches** null dates
- `greater_than` / `less_than` **never match** null numbers
- Array operators **never match** null arrays
- Null enrichment data → rules referencing those fields are skipped for that item
